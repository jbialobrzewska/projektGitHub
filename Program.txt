    private void jButton_seletcActionPerformed(java.awt.event.ActionEvent evt) {                                               
      if (buttonGroup.getSelection() == null)
        {
            JOptionPane.showMessageDialog(null, "Wybierz kamień papier bądź nożyce");
        }
        else
        {
            int computer = ComputerPlays();
            int player = PlayerPlays();
                        
            if (DecideWinner(computer, player) == 1) //komputer wygrywa
            {
                int computerScore = Integer.parseInt(jLabel_computerScore.getText());
                jLabel_computerScore.setText(Integer.toString(computerScore + 1));
                jLabel_computersSelection.setText(jLabel_computersSelection.getText() + " Komputer wygrał!");
            }

            else if (DecideWinner(computer, player) == 2) //gracz wygrywa
            {
                int playerScore = Integer.parseInt(jLabel_playerScore.getText());
                jLabel_playerScore.setText(Integer.toString(playerScore + 1));
                jLabel_computersSelection.setText(jLabel_computersSelection.getText() + " Ty Wygrałeś!");
            }
            else //remis
            {
                jLabel_computersSelection.setText(jLabel_computersSelection.getText() + " To Remis!");
            }
        }
    }

    // funkcja losujaca
    //1 -> kamień; 2 -> papier; 3 -> nozyce
    private int ComputerPlays()
    {
        Random rnd = new Random();
        int selection = (rnd.nextInt(3) + 1);
        
        if (selection == 1)
            jLabel_computersSelection.setText("Komputer wybrał: kamień");
        if (selection == 2)
            jLabel_computersSelection.setText("Komputer wybrał: papier.");
        if (selection == 3)
            jLabel_computersSelection.setText("Komputer wybrał: nożyce");
        
        return selection;
    }
